{

	no=b;
	while(b)                    //converting into decimal of the digits left of decimal by multipling with radix no. with its power increasing eac time
	{int a=fmod(b,10.0);
	d=d+p*a;
	b=b/10;
	p=p*i;	
	}
	 q=i;
	 b=no;
	while(b!=((int)b))               //converting after decimal point diving the digits with its power decreasing each time
	{
		b=b*10;
	int	z=fmod(b,10.0);
		d=d+(z/q);
		cout<<d;
		q*=i;
	}
	
	return d;
}

2. Conversion from base 10 to base r.

#include<iostream>
#include<math.h>
using namespace std;
double ten_to_r(int i,double b)
{
	double d=0,p=1,no,q=1,dp=0,ans=0;
	no=b;
	while(b>=1)                //converting into radix dits no. before decimal point by dividing and storing its remainder
	{int a=fmod(b,i);
	d=d+p*a;
	b=b/i;
	p=p*10;	
		}
	
	 b=no;
	 int j=1;
	 p=1;
	while(j<4)                 //converting into radix digits after decimal point by multipling and storing its digits left of decimal point
	{
	int	x=b;
		b=b-x;
	   b=b*i;
	   x=fmod(b,10.0);
	   dp=dp*p+x;
	   p=10;
	   j++;
	}
	dp=dp/1000;                //suming both the results
	
	ans=(d+dp);
	return ans;
	
}
int main()
{double n;
int  r;
	cout<<"Enter base point ";
	cin>>r;
	cout<<"Enter the no.";
	cin>>n;
	cout<<"No. in radix "<<r<<" is "<<ten_to_r(r,n);
	return 0;
}
Output :-

Enter the no.12.35
No. in radix 2 is 1100.01
